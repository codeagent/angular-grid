Date Range Picker

  
     angular
       .module('app', ['angular-grid-view'])
       .run(function($rootScope) {
          $rootScope.dateRange = {
              locale: {
                 format: 'lll'
              },
              startDate: moment().subtract(1, 'months'),
              endDate: moment(),
          };

          $rootScope.$watch('dateRange', function(v) {
              $rootScope.startDate = moment(v.startDate).format('DD.MM.YYYY');
              $rootScope.endDate = moment(v.endDate).format('DD.MM.YYYY');
          }, true);
       });
